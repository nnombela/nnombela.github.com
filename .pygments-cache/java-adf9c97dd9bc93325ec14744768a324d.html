static Object forEach(Collection <Object> collection, Closure closure) {
    Object[] array = collection.toArray();
    for (int i = 0; i <array.length; ++i) {
        Map<String, Object> args = new HashMap<String, Object>();
        args.put("index", i);
        args.put("it", array[i]);

        closure.execute(args);
    }
    return collection;
}
